-------Use GIT as local Version Control System-------

1. mkdir project_dir , cd project_dir
2. git init
3. touch index.html
4. git status
5. git add index.html
6. git commit -m "first commit"
7. nano index.html -> <HTML> , touch info.txt
8. git status
9. touch .gitignore , nano .gitignore  -> .*.txt
10. git status 
11. git add index.html , git commit -m "Second commit"
12. git log
13. nano index.html -> <HEAD> 
14. git restore index.html
15. nano index.html -> <HEAD>
16. git add index.html
17. git restore --staged index.html
18. git config --global alias.my-add add
19. git my-add index.html , git commit -m "Third commit"
20. git reset HEAD~
